#!/bin/bash # Manage Messages: Add or Remove message pages for Fanshawe FWB Church echo "====================================" echo " Fanshawe FWB Messages Manager" echo "====================================" echo "1) Add a message" echo "2) Remove a message" read -p "Select an option (1 or 2): " OPTION case "$OPTION" in 1) # ------------------------ # ADD MESSAGE # ------------------------ read -p "Enter message title (use lowercase, hyphen for spaces): " MSG read -p "Enter speaker name: " SPEAKER read -p "Enter type (Sermon / Devotional): " TYPE read -p "Enter scripture reference: " SCRIPTURE read -p "Enter a short summary: " SUMMARY read -p "Optional: paste video embed code or leave blank: " VIDEO_EMBED read -p "Optional: notes PDF file path (relative to mysite folder) or leave blank: " NOTES_PDF read -p "Key takeaways (1-3 sentences): " KEY_POINTS read -p "Reflection questions (optional): " REFLECTION DATE=$(date +%Y-%m-%d) FOLDER="messages/$DATE-$MSG" mkdir -p "$FOLDER" HTML_FILE="$FOLDER/index.html" # Copy template and replace placeholders cat messages/template.html \ | sed "s|a|$MSG|g" \ | sed "s|2026-02-22|$DATE|g" \ | sed "s|a|$SPEAKER|g" \ | sed "s|a|$TYPE|g" \ | sed "s|a|$SCRIPTURE|g" \ | sed "s|a|$SUMMARY|g" \ | sed "s||$VIDEO_EMBED|g" \ | sed "s||$NOTES_PDF|g" \ | sed "s|a|$KEY_POINTS|g" \ | sed "s|a|$REFLECTION|g" \ > "$HTML_FILE" echo "✅ Created message page: $HTML_FILE" # Add link to main messages/index.html LINK="
  • $MSG
    • " MAIN_INDEX="messages/index.html" if [ ! -f "$MAIN_INDEX" ]; then # Create main index if it doesn't exist echo " Messages - Fanshawe FWB

    Messages

      $LINK
    " > "$MAIN_INDEX" else # Add link if not already in index grep -qxF "$LINK" "$MAIN_INDEX" || sed -i "/<\/ul>/i $LINK" "$MAIN_INDEX" fi # Git add, commit, push git add . git commit -m "Add message: $MSG" git push origin master echo "🎉 Done! Message '$MSG' added and pushed." ;; 2) # ------------------------ # REMOVE MESSAGE # ------------------------ read -p "Enter message title to remove (use lowercase, hyphen for spaces): " MSG # Find and remove folder FOLDER_PATTERN="messages/*-$MSG" FOLDER_TO_DELETE=$(ls -d $FOLDER_PATTERN 2>/dev/null) if [ -n "$FOLDER_TO_DELETE" ]; then rm -rf "$FOLDER_TO_DELETE" echo "✅ Removed folder: $FOLDER_TO_DELETE" else echo "⚠ No folder found for '$MSG'" fi # Remove link from messages/index.html MAIN_INDEX="messages/index.html" if [ -f "$MAIN_INDEX" ]; then sed -i "\|$MSG|d" "$MAIN_INDEX" echo "✅ Removed link from messages/index.html" fi # Git add, commit, push git add . git commit -m "Remove message: $MSG" git push origin master echo "🎉 Done! Message '$MSG' removed and changes pushed." ;; *) echo "❌ Invalid option. Exiting." ;; esac
